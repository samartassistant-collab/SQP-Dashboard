function safeN(v){return(v!==null&&v!==undefined&&!isNaN(Number(v)))?Number(v):0;}

function sheetRows(ws){if(!ws)return[];var r=XLSX.utils.decode_range(ws['!ref']||'A1:K50');var rows=[];for(var ri=r.s.r;ri<=r.e.r;ri++){var row=[];for(var ci=r.s.c;ci<=r.e.c;ci++){var cell=ws[XLSX.utils.encode_cell({r:ri,c:ci})];row.push(cell?cell.v:null);}rows.push(row);}return rows;}

function findArea(rows,name){for(var i=0;i<rows.length;i++)for(var j=0;j<rows[i].length;j++)if(rows[i][j]&&String(rows[i][j]).toLowerCase().includes(name.toLowerCase()))return rows[i];return null;}

function parseArea(rows,name,tc,ac){var r=findArea(rows,name)||[];return{t:safeN(r[tc])||0,a:safeN(r[ac])||0};}


function parseMonthlyForm(wb,mi){

  hi.total.t=hi.internal.t+hi.outsourcing.t; hi.total.a=hi.internal.a+hi.outsourcing.a;


  var mkt={africa:{t:safeN((findArea(p6,'Africa')||[])[3])||100000,a:safeN((findArea(p6,'Africa')||[])[4])},me:{t:safeN((findArea(p6,'Middle East')||[])[3])||66667,a:safeN((findArea(p6,'Middle East')||[])[4])},total:{t:0,a:0}};

  mkt.total.t=mkt.africa.t+mkt.me.t; mkt.total.a=mkt.africa.a+mkt.me.a;


  function cashArea(rows,name){var r=findArea(rows,name)||[];return{dso:safeN(r[7])||0,ov90:safeN(r[4])||0,top5:safeN(r[6])||0,dues:safeN(r[8])||0};}

  var cash={ci:cashArea(p7,'CI'),cam:cashArea(p7,'CAM'),egypt:cashArea(p7,'Egypt'),global:{dso:0,dues:0}};

  cash.global.dso=Math.round((cash.ci.dso+cash.cam.dso+cash.egypt.dso)/3);

  cash.global.dues=cash.ci.dues+cash.cam.dues+cash.egypt.dues;

  var bl={ci:blCI,cam:blCAM,egypt:blEG,uae:0,me:0,global:blCI+blCAM+blEG};


  return{label_en:months_en[mi]+' 2026',label_fr:months_fr[mi]+' 2026',source:'excel-upload',pipeline:pip,revenue:rev,tcv:tcv,sga:sga,hirings:hi,market:mkt,cash:cash,backlog:bl};

}

function getVal(id){ var el=document.getElementById('pf-'+id); return el?el.value.trim():''; }

function getNum(id){ var v=parseFloat(getVal(id)); return isNaN(v)?0:v; }


function submitForm() {

  // Validate required fields

  var period = getVal('period');

  var name = getVal('name');

  var email = getVal('email');

  var status = getVal('status');


  if(!period||!name||!email||!status) {

    alert(lang==='fr'?'Merci de remplir tous les champs obligatoires (*)':'Please fill in all required fields (*)');

    return;

  }


  // Build submission object

  var sub = {

    id: Date.now(),

    pulse: currentPulse,

    period: period,

    name: name,

    email: email,

    status: status,

    msg: getVal('msg'),

    ts: new Date().toLocaleString(),

    lang: lang

  };


  // Pulse-specific fields

  if(currentPulse==='P1'){ sub.area=getVal('area'); sub.target=getNum('target'); sub.achieved=getNum('achieved'); }

  else if(currentPulse==='P2'){ sub.area=getVal('area'); sub.target=getNum('target'); sub.achieved=getNum('achieved'); sub.backlog=getNum('backlog'); }

  else if(currentPulse==='P3'){ sub.area=getVal('area'); sub.target=getNum('target'); sub.achieved=getNum('achieved'); sub.deals=getNum('deals'); }

  else if(currentPulse==='P4'){ sub.area=getVal('area'); sub.people_cost=getNum('people_cost'); sub.nonpeople_cost=getNum('nonpeople_cost'); sub.revenue=getNum('revenue'); sub.total_cost=sub.people_cost+sub.nonpeople_cost; sub.sga_rate=sub.revenue>0?sub.total_cost/sub.revenue:0; }

  else if(currentPulse==='P5'){ sub.internal=getNum('internal'); sub.outsourcing=getNum('outsourcing'); sub.filled=getVal('filled'); sub.open=getVal('open'); sub.blockers=getVal('blockers'); }

  else if(currentPulse==='P6'){ sub.region=getVal('region'); sub.target=getNum('target'); sub.achieved=getNum('achieved'); sub.solution=getVal('solution'); sub.new_client=getVal('new_client'); }

  else if(currentPulse==='P7'){ sub.area=getVal('area'); sub.dso=getNum('dso'); sub.overdue90_eur=getNum('overdue90_eur'); sub.overdue90_pct=getNum('overdue90_pct'); sub.top5_eur=getNum('top5_eur'); sub.top5_pct=getNum('top5_pct'); sub.total_dues=getNum('total_dues'); sub.actions=getVal('actions'); }

  else if(currentPulse==='P8'){ sub.region=getVal('region'); sub.country=getVal('country'); sub.consulting=getNum('consulting'); sub.licencing=getNum('licencing'); sub.won_to_be=getNum('won_to_be'); sub.total=sub.consulting+sub.licencing+sub.won_to_be; }


  // Save to localStorage

  submissions.push(sub);

  localStorage.setItem('sqp_submissions', JSON.stringify(submissions));


  // Update dashboard data

  applySubmissionToDashboard(sub);


  // Show success screen with email option

  showSuccess(sub);

}


function applySubmissionToDashboard(sub) {

  var d = getData();

  if(!d) return;

  var p = sub.period;

  // Try to find matching period and update

  if(sub.pulse==='P1' && sub.area && d.pipeline[sub.area.toLowerCase()]) {

    if(sub.target) d.pipeline[sub.area.toLowerCase()].t = sub.target;

    if(sub.achieved) d.pipeline[sub.area.toLowerCase()].a = sub.achieved;

    // Recalc global

    var keys=['ci','cam','egypt','uae']; var gt=0,ga=0;

    keys.forEach(function(k){gt+=d.pipeline[k].t;ga+=d.pipeline[k].a;});

    d.pipeline.global.t=gt; d.pipeline.global.a=ga;

  }

  if(sub.pulse==='P2' && sub.area && d.revenue[sub.area.toLowerCase()]) {

    if(sub.target) d.revenue[sub.area.toLowerCase()].t = sub.target;

    if(sub.achieved) d.revenue[sub.area.toLowerCase()].a = sub.achieved;

    if(sub.backlog!==undefined) d.revenue[sub.area.toLowerCase()].bl = sub.backlog;

    var keys=['ci','cam','egypt']; var gt=0,ga=0,gb=0;

    keys.forEach(function(k){gt+=d.revenue[k].t;ga+=d.revenue[k].a;gb+=d.revenue[k].bl;});

    d.revenue.global.t=gt; d.revenue.global.a=ga; d.revenue.global.bl=gb;

  }

  if(sub.pulse==='P3' && sub.area && d.tcv[sub.area.toLowerCase()]) {

    if(sub.target) d.tcv[sub.area.toLowerCase()].t = sub.target;

    if(sub.achieved!==undefined) d.tcv[sub.area.toLowerCase()].a = sub.achieved;

    var keys=['ci','cam','egypt','uae']; var gt=0,ga=0;

    keys.forEach(function(k){gt+=d.tcv[k].t;ga+=d.tcv[k].a;});

    d.tcv.global.t=gt; d.tcv.global.a=ga;

  }

  if(sub.pulse==='P4' && sub.area && d.sga[sub.area.toLowerCase()]) {

    d.sga[sub.area.toLowerCase()].a = sub.sga_rate||0;

    var keys=['ci','cam','egypt','uae']; var tc=0,tr=0;

    // Can't perfectly recalc without all costs, just update the area rate

  }

  if(sub.pulse==='P5') {

    if(sub.internal!==undefined) d.hirings.internal.a = sub.internal;

    if(sub.outsourcing!==undefined) d.hirings.outsourcing.a = sub.outsourcing;

    d.hirings.total.a = d.hirings.internal.a + d.hirings.outsourcing.a;

  }

  if(sub.pulse==='P6') {

    var key = (sub.region||'').toLowerCase().includes('afr')||((sub.region||'').toLowerCase().includes('afrique')) ? 'africa' : 'me';

    if(sub.target) d.market[key].t = sub.target;

    if(sub.achieved) d.market[key].a = sub.achieved;

    d.market.total.t = d.market.africa.t + d.market.me.t;

    d.market.total.a = d.market.africa.a + d.market.me.a;

  }

  if(sub.pulse==='P7' && sub.area) {

    var key = sub.area.toLowerCase().replace('é','e');

    if(d.cash[key]) {

      if(sub.dso) d.cash[key].dso = sub.dso;

      if(sub.overdue90_pct!==undefined) d.cash[key].ov90 = sub.overdue90_pct/100;

      if(sub.top5_pct!==undefined) d.cash[key].top5 = sub.top5_pct/100;

      if(sub.total_dues) d.cash[key].dues = sub.total_dues;

      var dsAvg = Math.round((d.cash.ci.dso+d.cash.cam.dso+d.cash.egypt.dso)/3);

      d.cash.global.dso = dsAvg;

    }

  }

  if(sub.pulse==='P8') {

    var ckey = (sub.country||'ci').toLowerCase().replace('é','e').replace('î','i').replace('û','u').replace(' ','_');

    if(ckey==='ci') d.backlog.ci = sub.total||0;

    else if(ckey==='egypt'||ckey==='égypte') d.backlog.egypt = sub.total||0;
